1 /**
  2  * @fileoverview the mutil.js tiny javascript library.
  3  * 
  4  * <h2>Introduction</h2>
  5  * 
  6  * <p>μtil (mutil) is a tiny Javascript utility library that looks a lot like 
  7  * jQuery but is much smaller. It has a simple selector engine, some basic 
  8  * event methods and a simple plugin architecture to easily add functions to 
  9  * its core.</p>
 10  * 
 11  * <h2>Example code</h2>
 12  * 
 13  * <p>Here's some demo code:</p>
 14  * 
 15  * <pre class="code">
 16  * // on page load
 17  * $(function() {
 18  * 
 19  *   // Select all links in the content leader
 20  *   $('#content p.leader a')
 21  * 
 22  *     // observe the click event
 23  *     .addEvent('click', function(e) {
 24  * 
 25  *       // stop event propagation and prevent default link behaviour
 26  *       e.stop();
 27  * 
 28  *       // Set a nice background color on its parent
 29  *       $(this.parentNode).setCss({ background: '#ffffe1' });
 30  *   });
 31  * });
 32  * </pre>
 33  * 
 34  * <h2>Plugins</h2>
 35  * 
 36  * <p>You can add extra functions to the μtil toolkit using plugins:</p>
 37  * 
 38  * <pre class="code">
 39  * // Use a closure to make sure $ is μtil in this context. 
 40  * (function($){
 41  * 
 42  *   // Declare a new function on the fn property of μtil
 43  *   // which will then be available on all collectionss
 44  *   $.fn.printer = function() {
 45  * 
 46  *     // Loop over all selected elements and return the iterator,
 47  *     // so we can chain methods
 48  *     return this.each(function() {
 49  * 
 50  *       // run the plugin code here: attach an event.
 51  *       $(this).addEvent('click', function(e) {
 52  *         e.stop();
 53  *         window.print();
 54  *       });
 55  *     });
 56  *   };
 57  * })(mutil);
 58  * </pre>
 59  * 
 60  * After loading this snippet in your page you can let any object open the
 61  * print dialog when clicked upon:
 62  * 
 63  * <pre class="code">
 64  * $('#print-link').printer();
 65  * </pre>
 66  * 
 67  * <h2>Progress</h2>
 68  * 
 69  * <p>This is still very much a work in progress. Here's whats on my 
 70  * todo-list:</p>
 71  * 
 72  * <ol>
 73  *   <li>Create a test suite</li>
 74  *   <li>Add some more useful plugins</li>
 75  *   <li>simplify and shorten the code</li>

 78  * <p>Here's what I don't want to:</p>

 83  * <li>Complete DOM traversal</li>

 85  * 
 86  * <p>When you're in need of that stuff, you're probably better off using
 87  * prototype or jQuery or some other major framework.</p>
 88  *

 96 /**
 97  * Empty constructor, since this is a mix-in module and not a real class.
 98  * @class The enumerable mix in provides helper functions for working with
 99  * arrays. It is mixed into the array prototype and is applied to all element
100  * collections.
101  * 
102  * @constructor
103  */
104 function Enumerable() {};
105 Enumerable.prototype = {
106 	/**
107 	 * Loop over the elements in an array, yielding each to the callback. The
108 	 * callback receives as arguments the index of iteration and the entire array.
109 	 * 
110 	 * Usage:
111 	 * 
112 	 * <pre class="code">
113 	 * ['one', 'two', 'three'].each(function(i, arr) {
114 	 *   console.log(this);
115 	 * });
116 	 * </pre>
117 	 * 
118 	 * Produces:
119 	 * 
120 	 * <pre class="code">
121 	 * one
122 	 * two
123 	 * three
124 	 * </pre>
125 	 * 
126 	 * @param {Function} callback is the function to call for every element.
127 	 * @returns itself
128 	 * @type Array
129 	 */
130 	each: function(callback) {
131 	    for(var i = 0, j = this.length; i < j; i++) {
132 			callback.call(this[i], i, this);
133 		}
134 		return this;
135 	},
136 
137 	/**
138 	 * Reduce an array to a single value with a callback.
139 	 * 
140 	 * Usage:
141 	 * 
142 	 * <pre class="code">
143 	 * [1, 2, 3].inject(0, function(output, i, arr) {
144 	 *   return output + this;
145 	 * });
146 	 * </pre>
147 	 * 
148 	 * Produces:
149 	 * 
150 	 * <pre class="code">
151 	 * 6
152 	 * </pre>
153 	 *
154 	 * @param init the starting value
155 	 * @param {Function} callback is the function to call for every element.
156 	 * @returns the cumulative result value
157 	 */
158 	inject: function(init, callback) {
159 		this.each(function(i, arr) {
160 			init = callback.call(this, init, i, arr);
161 		});
162 		return init;
163 	},
164 
165 	/**
166 	 * Create a new array containing those elements from the original array for
167 	 * which the callback return value evaluates to true.
168 	 * 
169 	 * Usage:
170 	 * 
171 	 * <pre class="code">
172 	 * [1, 2, 3, 4].select(function(i, arr) {
173 	 *   return this > 2;
174 	 * });
175 	 * </pre>
176 	 * 
177 	 * Produces:
178 	 * 
179 	 * <pre class="code">
180 	 * [3, 4]
181 	 * </pre>
182 	 * 
183 	 * @param {Function} callback to determine to select an element or not
184 	 * @returns a new array with only the selected elements
185 	 * @type Array
186 	 */
187 	select: function(callback) {
188 		return this.inject([], function(memo, i, arr) {
189 			if(callback.call(this, i, arr)) memo.push(this);
190 			return memo;
191 		});
192 	},
193 
194 	/**
195 	 * Returns if the array has any elements. If the optional callback is given,
196 	 * it will return the count of elements that evaluate to true for the
197 	 * callback.
198 	 * 
199 	 * Usage:
200 	 * 
201 	 * <pre class="code">
202 	 * [].any();
203 	 * [1, 2, 3].any();
204 	 * [1, 2, 3].any(function() {
205 	 *   return this > 2;
206 	 * });
207 	 * [1, 2, 3].any(function() {
208 	 *   return this > 4;
209 	 * });
210 	 * </pre>
211 	 * 
212 	 * Produces:
213 	 * 
214 	 * <pre class="code">
215 	 * false
216 	 * true
217 	 * true
218 	 * false
219 	 * </pre>
220 	 * 
221 	 * @param {Function} callback to select which elements to count (optional)
222 	 * @returns whether there are any elements
223 	 * @type Bool
224 	 */
225 	any: function(callback) {
226 		if(typeof callback == 'function') {
227 			return count(this.select(callback)) > 0;
228 		} else {
229 			return count(this) > 0;
230 		}
231 	},
232 
233 	/**
234 	 * Apply a callback function to every element in the array.
235 	 * 
236 	 * Usage:
237 	 * 
238 	 * <pre class="code">
239 	 * [1,2,3].map(function(i, arr) {
240 	 *   return this * 2;
241 	 * });
242 	 * </pre>
243 	 * 
244 	 * Produces:
245 	 * 
246 	 * <pre class="code">[2,4,6];</pre>
247 	 * 
248 	 * @param {Function} callback to apply to every element
249 	 * @returns a new array with callback applied to every element
250 	 * @type Array
251 	 */
252 	map: function(callback) {
253 		return this.inject([], function(memo, i, arr) {
254 			return memo.push(callback.call(this, i, arr));
255 		});
256 	}
257 };
258 for(method in Enumerable.prototype) {
259 	if(!Array.prototype[method]) Array.prototype[method] = Enumerable.prototype[method];
260 }

263 /**
264  * Remove leading and trailing whitespace.
265  * 
266  * Usage:
267  * 
268  * <pre class="code">" my string   ".trim()</pre>
269  * 
270  * Produces:
271  * 
272  * <pre class="code">"my string"</pre>
273  * 
274  * @returns itself with leading and trailing whitespace removed
275  * @type String
276  */
277 String.prototype.trim = function() {
278 	return this.replace(/^\s\s*/, '').replace(/\s\s*$/, '');
279 };

281 /**
282  * Create a new mutil object, selecting either some DOM elements or adding
283  * and onload event in the process.
284  * 
285  * @class Select DOM elements by using a really simple CSS selector syntax.
286  * 
287  * Supported selectors:
288  * 
289  * <ol>
290  *   <li>by ID: <code>#content</code></li>
291  *   <li>by tag name: <code>div</code></li>
292  *   <li>by descendant: <code>div p</code></li>
293  *   <li>by class name: <code>p.test</code></li>
294  * </ol>
295  * 
296  * You can combine these selectors to your liking, i.e. '#content div p.test'.
297  * 
298  * If you pass in an existing DOM element or Nodelist, that is returned
299  * (the result is always an array).
300  * 
301  * Usage:
302  * 
303  *   <pre class="code">$('#content p.test');</pre>
304  * 
305  * Produces:
306  * 
307  *   <pre class="code">[p.test, p.test]</pre>
308  * 
309  * In addition you can pass in a function as an argument and have it fire
310  * when the onLoad-event fires.
311  * 
312  * Usage:
313  * 
314  * <pre class="code">
315  * $(function() {
316  *   $('p').addEvent(...);
317  * });
318  * </pre>
319  * 
320  * @constructor
321  * @requires Enumerable
322  * @param arg A simple CSS selector, DOM element or Nodelist.
323  * @param root is a DOM element in whose children to search for arg.
324  * @augments Enumerable
325  * @borrows Enumerable#each as #each
326  */
327 function Mutil(arg, root) {
328 	/**
329 	 * Select DOM elements helper function.
330 	 * 
331 	 * @inner
332 	 */
333 	var find = function(arg, root) {
334 		
335 		// Helper method to convert a nodelist to an array
336 		this.nodelistToArray = function(nodelist) {
337 			var arr = [];
338 			for(var i = 0, j = nodelist.length; i < j; i++) { arr.push(nodelist[i]); }
339 			return arr;
340 		};
341 		
342 		// Set the root element to select from. Default to document.
343 		// If there are multiple roots run this function that many times.
344 		if(typeof root == 'undefined') root = document;
345 		if(root == null) return [];
346 		if(typeof root.length != 'undefined') {
347 			return root.inject([], function(output) {
348 				return output.concat($(arg, this));
349 			});
350 		}
351 
352 		// Select by NodeList: return a real array of elements
353 		if(typeof arg.length != 'undefined' && typeof arg.item != 'undefined') return this.nodelistToArray(arg);
354 		
355 		// Select by Node: return a node element
356 		if(typeof arg.nodeType != 'undefined') return [arg];
357 
358 		// Select by ID: return the element
359 		if(matches = arg.match(/^#([a-zA-Z0-9\-_]+)$/)) {
360 			return root.getElementById(matches[1]);
361 		}
362 
363 		// Select multiple: find selectors seperated by commas
364 		if(arg.match(/,/)) {
365 			return arg.split(',').inject([], function(output) {
366 				return output.concat($(this.trim()));
367 			});
368 		}
369 
370 		// Select by descendent: find elements inside other elements
371 		// #foo bar table.test
372 		if(arg.match(/ /)) {
373 			var parts = arg.split(' '),
374 			    parent = parts.shift(),
375 			    selector = parts.join(' ');
376 			return $(selector, $(parent, root));
377 		}
378 
379 		// Select by classname: split by dot and find all elements, filter by the rest
380 		if(arg.match(/[#a-zA-Z0-9\-_]+\.[a-zA-Z0-9\-_.]+/)) {
381 			var parts = arg.split('.'),
382 			    tagname = parts.shift(),
383 			    classes = new RegExp(parts.join('|'), 'gi');
384 			return $(tagname, root).select(function() {
385 				return this.className.match(classes);
386 			});
387 		}
388 
389 		// Select by tagname: return array of elements
390 		return this.nodelistToArray(root.getElementsByTagName(arg));
391 	};
392 	
393 	/**
394 	 * Register a load event function helper function.
395 	 * 
396 	 * @inner
397 	 */
398 	var addLoadEvent = function(func) {
399 		var oldonload = window.onload;
400 		if (typeof window.onload != 'function') {
401 			window.onload = func;
402 		} else {
403 			window.onload = function() {
404 				if (oldonload) oldonload();
405 				func();
406 			};
407 		}
408 	};
409 	
410 	if(typeof arguments[0] == 'function') {
411 		addLoadEvent(arguments[0]);
412 	} else {
413 		this.elements = find(arguments[0], arguments[1]);
414 	}
415 }

417 Mutil.prototype = {
418 	/*
419 	 * Import Enumerable functions
420 	 * 
421 	 * TODO: clean this up somehow.
422 	 */
423 	each:   function() { return Enumerable.prototype.each  .apply(this.elements, arguments); },
424 	inject: function() { return Enumerable.prototype.inject.apply(this.elements, arguments); },
425 	select: function() { return Enumerable.prototype.select.apply(this.elements, arguments); },
426 	map:    function() { return Enumerable.prototype.map   .apply(this.elements, arguments); },
427 	any:    function() { return Enumerable.prototype.any   .apply(this.elements, arguments); },
428 	
429 	/**
430 	 * Add an event to a DOM element.
431 	 * 
432 	 * Usage:
433 	 * 
434 	 * <pre class="code">
435 	 * addEvent(document.getElementById('test'), 'click', function(e) {
436 	 *   console.log('clicked!');
437 	 * });
438 	 * </pre>
439 	 * 
440 	 * You can also use this method as an array function, applying it to
441 	 * all elements in an array.
442 	 * 
443 	 * The single argument to the callback function is an {@link Event} object
444 	 * that works just like the native event object, but is cross-browser
445 	 * compatible.
446 	 * 
447 	 * @see Event
448 	 * @param {String} type is the event to observe.
449 	 * @param {Function} callback is the observer function.
450 	 * @returns itself for chaining purposes.
451 	 * @type Mutil
452 	 */
453 	addEvent: function(type, callback) {
454 		function FwEvent(e) {
455 			var base = this;
456 			this.event = e || window.event;
457 			this.target = this.event.target || this.event.srcElement;
458 			this.stopPropagation = function() {
459 				e.stopPropagation ? e.stopPropagation() : e.cancelBubble = true;
460 			};
461 			this.preventDefault = function() {
462 				e.preventDefault ? e.preventDefault() : e.returnValue = false;
463 			};
464 			this.stop = function() {
465 				this.stopPropagation();
466 				this.preventDefault();
467 			};
468 		}
469 		var add = function(el, fn) {
470 			if(window.addEventListener) {
471 				el.addEventListener(type, fn, false);
472 			} else if(window.attachEvent) {
473 				el.attachEvent('on' + type, fn);
474 			} else {
475 				el['on' + type] = fn;
476 			}
477 		};
478 		this.each(function() {
479 			add(this, function(e) { callback.call(this, new FwEvent(e)); });
480 		});
481 		return this;
482 	},
483 	
484 	/**
485 	 * Give an element a class.
486 	 * 
487 	 * @example
488 	 * $('p').addClass('highlight');
489 	 * 
490 	 * @param {String} value is the class name to add
491 	 * @returns itself for chaining purposes
492 	 * @type Mutil
493 	 */
494 	addClass: function(value) {
495 		return this.each(function() {
496 			if(!this.className)
497 				this.className = value;
498 			else if(!this.className.match(new RegExp('(\\s|^)' + value + '(\\s|$)')))
499 				this.className += ' ' + value;
500 		});
501 	},
502 	
503 	/**
504 	 * Remove an element's class.
505 	 * 
506 	 * @example
507 	 * $('p').removeClass('highlight');
508 	 * 
509 	 * @param {String} value is the class name to remove
510 	 * @returns itself for chaining purposes
511 	 * @type Mutil
512 	 */
513 	removeClass: function(value) {
514 		return this.each(function() {
515 			this.className = this.className.replace(new RegExp('(\\s|^)' + value + '(\\s|$)'), '');
516 		});
517 	},
518 	
519 	/**
520 	 * Tell if an element has a given class.
521 	 * 
522 	 * When working on multiple elements this returns whether ALL the elements
523 	 * have the given class.
524 	 * 
525 	 * @example
526 	 * if($('p').hasClass('highlight')) { ... }
527 	 * 
528 	 * @param {String} value is the class name to add
529 	 * @returns whether the element(s) has the class or not
530 	 * @type Boolean
531 	 */
532 	hasClass: function(value) {
533 		return this.inject(false, function(m) {
534 			return m || this.className.match(new RegExp('(\\s|^)' + value + '(\\s|$)'));
535 		});
536 	},
537 	
538 	/**
539 	 * Switch an element's class on or off.
540 	 * 
541 	 * Either remove or add the class to the element, depending on
542 	 * whether the element already has the class.
543 	 * 
544 	 * @param {String} value the class name to switch on or off
545 	 * @returns itself for chaining purposes
546 	 * @type Mutil
547 	 */
548 	toggleClass: function(value) {
549 		return this.each(function() {
550 			$(this).hasClass(value) ? $(this).removeClass(value) : $(this).addClass(value);
551 		});
552 	},
553 	
554 	/**
555 	 * Give an element an arbitrary set of CSS properties.
556 	 * 
557 	 * @example
558 	 * $('p').setCss({ color: '#f00' });
559 	 * 
560 	 * @param {Object} key/value object of CSS attributes and values.
561 	 * @returns itself for chaining purposes
562 	 * @type Mutil
563 	 */
564 	setCss: function(styles) {
565 		return this.each(function() {
566 			for(var prop in styles) {
567 				if(!styles.hasOwnProperty(prop)) continue;
568 				this.style[prop] = styles[prop];
569 			}
570 		});
571 	}
572 };
